r - any register name
i - any const immediate
l - label
b - r or b

False = 0
True = cell_size^(num_size-1)
False and True are immediate aliases

Copy:
mov r b
load r b
store b b

Math:
neg r
add r b b
sub r b b
less r b b    - Unsigned comparision
eq r b b

Logic:
logic_not r b
logic_and r b b
logic_or r b b

Binary:
binary_not r b

Stack:
push b
pop r

Flow:
ret
label l
jmp l
jnz r l
call l    :=   push next_basic_block; goto label;
