r - any register name
i - any const immediate
b - r or b

False = 0
True = cell_size^(num_size-1)
False and True are immediate aliases

Copy:
r = b
r = *b
*b = b

Math:
r = b + b
r = b - b
r = b < b   - Unsigned comparision
r = b <= b  - Unsigned comparision
r = b > b   - Unsigned comparision
r = b >= b  - Unsigned comparision
r = b == b
r = b != b

Logic:
r = !r
r = r && r
r = r || r

Binary:
r =~ b

Stack:
push b
pop r
ret

Flow:
jmp label
label:
if r, endif
if r, else, endif
while r, endwhile
call label    :=   push next_basic_block; goto label;
